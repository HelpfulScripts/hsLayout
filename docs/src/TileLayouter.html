  42## Attributes
  43The following values **v** are valid entries in The Attributes array:
  44- **px(n)** or **"_n_ px"** -- a fixed number of pixels 
  45- **pc(n)** or **"_n_ %"**  -- a fixed percentage of available space
  46- **FILL** or **"fill"**   -- a special constant indicate to fill remaining space - may appear only once per array.
  47
  48The following options are supported for the Attributes array:
  49- **[ ]**: An empty array; The available tiles will cover the entire width and height. 
  50    Their size will adapt as tiles are added.
  51- **[FILL]**: like [ ], except the last tile fills the available space. 
  52- **[v]**: All components have the specified width and height (in px or %) and will fill the available space from the left,
  53    leaving any remaining unused space on the right. 
  54- **[w, h]**: All components have the specified width w and height h (in px or %) and will fill the available space from the left,
  55    leaving any remaining unused space on the right. 
  56- **[v, FILL]**: like [v], except that the last tile in each row will fill the remaining available width 
  57    and the tiles in the bottom row will fill the remaining height
  58- **[w, h, FILL]**: like [w, h], except that the last tile will fill the remaining available width. 
  59 */

  73class Tiles extends Layouter {
  74    cssClass:string;

  77    /**
  78     * Constructs a Tileds layout
  79     * @param areaDesc Description of the requested layout 
  80     */
  81    constructor(public areaDesc:LayoutToken[]) { 
  82        super(areaDesc); 
  83        // if any of the dimensions are in px, use the pixel method; else use the percent method
  84        // get unitPixel if any area is PixelToken           
  85        this.unit = areaDesc.some((area:LayoutToken) => (area instanceof PixelToken))? 
  86            this.unitPixel : this.unitPercent;           
  87    };
  88
  89    private unitPercent(num:number) {
  90        const desc = this.areaDesc;
  91        const fill = this.areaDesc.some(a => (a instanceof FillToken));
  92        const root = Math.sqrt(num);
  93        const rows = Math.round(root);
  94        let   cols = Math.floor(root);
  95        if (root > cols) { cols++; }
  96        let width  = (desc[0] instanceof DefinedToken)? desc[0].getSize() : undefined;
  97        let height = (desc[1] instanceof DefinedToken)? desc[1].getSize() : width;
  98
  99        width  = width  || 100/cols;
 100        height = height || 100/rows;
 101        let left = 0;
 102        let top  = 0;
 103
 104        let styles = [...Array(num).keys()].map(i => { 
 105            let r = 'auto';    let w = width+'%'; 
 106            let b = 'auto';    let h = height+'%';
 107            if ((left + 2*width) > 100 && fill) { r = '0%'; w = 'auto'; }
 108            if ((top + 2*height) > 100 && fill) { b = '0%'; h = 'auto'; }
 109            const style = `
 110                top: ${Math.floor(top)}%; bottom:${b};
 111                left: ${left}%;           right:${r};
 112                width: ${w};              height: ${h};
 113            `;
 114            if (Math.round(left += width) > 100-Math.floor(width)) { left = 0; top += height; }
 115            return style;
 116         });
 117        return styles;    
 118    };
 119
 120    private unitPixel(num:number) { // pattern: [px, px, FILL]
 121        const desc = this.areaDesc;
 122//        const fill = this.areaDesc.some(a => (a instanceof FillToken));
 123        const root = Math.sqrt(num);
 124        const rows = Math.round(root);
 125        let   cols = Math.floor(root);
 126        if (root > cols) { cols++; }
 127        let width  = (desc[0] instanceof DefinedToken)? desc[0].getSize() : undefined;
 128        let height = (desc[1] instanceof DefinedToken)? desc[1].getSize() : width;
 129
 130        width  = width  || 100/cols;
 131        height = height || 100/rows;
 132        let left = 0;
 133        let top  = 0;
 134
 135        let styles = [...Array(num).keys()].map(i => { 
 136            let r = 'auto';    let w = width+'px'; 
 137            let b = 'auto';    let h = height+'px';
 138            const style = `
 139                top: ${Math.floor(top)}%; bottom:${b};
 140                left: ${left}%;           right:${r};
 141                width: ${w};              height: ${h};
 142            `;
 143            if (Math.round(left += width) > 100-Math.floor(width)) { left = 0; top += height; }
 144            return style;
 145         });
 146        return styles;    
 147    };
 148    
 149    /**
 150     * Calculates the style attributes required for each component in `Components`.
 151     * These attributes are saved in a `styles` field on the component itself. 
 152     * During rendering these `styles` attributes are copied to the `node.attrs.styles` field.
 153     * @param components 
 154     */
 155    protected getStyles(components:Array):string  { 
 156        let styles = this.unit(components.length);
 157        components.map((c:Layout|Vnode, i:number) => {
 158            c.style = styles[i];
 159        });   
 160        return '.hs-tile-layout';
 161    };
