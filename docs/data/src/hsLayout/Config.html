   1/**

   2 * ## Config 

   3 * Tool to configure a tree of mithril components via an object literal or as a JSON file.

   4 * The object format is `{ : { : }, ...}`.

   5 * The name `root` is required and defines the top of the tree. Other named sections are allowed and

   6 * can be referenced by name inside the `root` structure, 

   7 * This can be used to structure the obect in a more readable and maintainable format.

   8 * 

   9 * ### Example:

  10 * In the example below, the comnponents `Layout`, `LeftNav`, `MainNav` will be resolved against modules

  11 * provided in the `context` parameter when creating the `Config` node. 

  12 * ```

  13 * {

  14 *  root: { Layout: {

  15 *      rows:  ["30px", "fill", "10px"],

  16 *      css: '.my-example',

  17 *      content: ['Header', 'Body', 'Footer]

  18 *  }},

  19 *  Header: { Layout:{

  20 *      columns: ["200px", "fill"],

  21 *      content: ['Left Header', 'Main Header']

  22 *  }},

  23 *  Body: { Layout:{

  24 *      columns: ["200px", "fill"], 

  25 *      content: ['LeftNav', 'MainNav']

  26 *  }},

  27 *  Footer: { Layout: {

  28 *      css: '.hs-site-footer',

  29 *      content: ['(c) Helpful ; Scripts']

  30 *  }},

  31 *  LeftNav:  { LeftNav:  { lib:"route.lib", field:"route.field"}},

  32 *  MainNav:  { MainNav:  { lib:"route.lib", field:"route.field"}}

  33 * }

  34 * ```

  35 * 

  36 * script.js'>

  37 * const theContent = {

  38 *      root: { Layout: {

  39 *          rows:  ["30px", "fill", "30px"],

  40 *          css: '.my-example',

  41 *          content: ['Header', 'Body', 'Footer']

  42 *      }},

  43 *      Header: { Layout:{

  44 *          columns: ["100px", "fill"],

  45 *          content: ['Left Header', 'Main Header']

  46 *      }},

  47 *      Body: { Layout:{

  48 *          columns: ["100px", "fill"], 

  49 *          content: ['LeftNav', 'MainNav']

  50 *      }},

  51 *      Footer: { Layout: {

  52 *          css: '.my-footer',

  53 *          content: ['(c) Helpful Scripts']

  54 *      }}

  55 * }

  56 * m.mount(root, {view: () => 

  57 *      m(hsLayout.Config, { 

  58 *          source: theContent, // a file name, or obect literal

  59 *          context:[hsLayout]  // a list of es6 modules against which to resolve components at runtime 

  60 *      })

  61 * });

  62 * 

  63 * 

  64 * 

  65 * ### Calling pattern

  66 * Instantiate the component via

  67 * ```

  68 *  m(Config, { 

  69 *      source: src, // a file name, or obect literal

  70 *      context:[]   // a list of es6 modules against which to resolve components at runtime 

  71 *  });

  72 * ```

  73 * Additional parameters can be passed to the root node. The example below establishes routing and passes

  74 * `field` and `lib` to the root node:

  75 * ```

  76 *  class Router {

  77 *      view(node:Vnode) { 

  78 *          return m(Config, Object.assign({source: src, context: [hsLayout]}, node.attrs));

  79 *      }

  80 *  }

  81 *  m.route(document.body, '/api', { 

  82 *      '/api':             Router,

  83 *      '/api/:lib':        Router,

  84 *      '/api/:lib/:field': Router

  85 *  });

  86 * ```

  87 */

  91import { Log }  from 'hsutil'; const log = new Log('Config');

  93/**

  94 * `Component` class that creates a tree of mithril components out of a configuration obect or file.

  95 */

  96export class Config {

  97    async oninit(node:Vnode) {

  98        const context:any[] = node.attrs.context;

  99        if (!node.state.cfg) {

 100            const s = (typeof node.attrs.source === 'string')?

 101                await m.request({ method: "GET", url: node.attrs.source})

 102              : node.attrs.source;

 103            node.state.cfg = translate(s, s.root, context);

 104        }

 105    }

 106    view(node:Vnode) { 

 107        const cfg = node.state.cfg;

 108        return (cfg && cfg.compClass)? m(cfg.compClass, Object.assign(Object.assign({}, cfg.attrs), node.attrs)) : m('div', 'waiting');

 109    }

 110}

 112/**

 113 * recursively translates a configuration, trying to fetch the class definition for each element (key) in `config`. 

 114 * If successful, it creates an object literal containing the component class and its attributes.

 115 * If unsuccessful, the element's value is returned unaltered so that it can be consumed 

 116 * by an instance further up in the recursion tree.

 117 * @param config an object literal containing the entire configuration tree

 118 * @param subcfg an object literal containing a configuration subtree

 119 * @param context an array of objects against which to instantiate elements of `config`.

 120 * @return an object literal representing the configuration, with Class names resolved 

 121 * against the provided `context`.

 122 */

 123function translate(config:any, subcfg:any, context:any[]) {

 124    // resolve if synonym

 125    if (isSynonym(config, subcfg)) { subcfg = config[subcfg]; }

 126    // return if primitive

 127    if (['string', 'number', 'boolean', 'function'].indexOf(typeof subcfg)>=0) { return subcfg; }

 128    let result = subcfg.length? [] : {};

 129    // step through options 

 130    const options = Object.keys(subcfg);

 131    options.map((opt:string):Vnode => {

 132        const cl:any = resolve(opt, context);

 133        const content = translate(config, subcfg[opt], context); 

 134        // if a class resolution exists:

 135        if (cl) { 

 136            log.debug(`resolved class '${opt}' to ${log.inspect(cl, 1)}`);

 137            const r = {

 138                compClass:cl,   // Component class

 139                attrs:content   // attributes passed to the Component class

 140            };

 141            (!Array.isArray(subcfg) && options.length === 1)? 

 142                result = r : 

 143                result[opt] = r;   

 144        }

 145        // otherwise, if no class resolution exists:

 146        else { 

 147            if (isNaN(parseInt(opt))) {

 148                log.debug(`resolved direct '${opt}' to ${log.inspect(content, 0)}`);

 149            }

 150            result[opt] = content; 

 151        }

 152    }); 

 153    return result; 

 154}

 156/**

 157 * resolves the symbol `sym` against the provided `context`.

 158 * If `context` has a key `sym`, returns `context[sym]` as the literal definition for `sym`. 

 159 * @param sym the symbol to resolve

 160 * @param context the context to resolve against; `mithril` and `hsLayout` 

 161 * are implicitely part of the context and need not be specified.

 162 * @return the resolved Class, or `undefined`.

 163 */

 164function resolve(sym:string, context:any[]) {

 165    log.debug(`resolving ${sym} in context '${log.inspect(context,0)}'`);

 166    let cl:any;

 167    context.some((c:any) =>  cl = c[sym]);

 168    log.debug(`resolving ${sym} => ${log.inspect(cl,0)}`);

 169    return cl;

 170}

 172function isSynonym(config:any, key:any) { return typeof key === 'string' && config[key]; }
