   1/**

   2 * ## Config 

   3 * Tool to configure a tree of mithril components via an object literal or as a JSON file.

   4 * The object format is `{ : { : }, ...}`.

   5 * The name `root` is required an defines the top of the tree. Other named sections are allowed and

   6 * can be used to structure the obect in a more readable and maintainable format.

   7 * 

   8 * ### Example:

   9 * In the example below, the comnponents `Layout`, `LeftNav`, `MainNav` will be resolved against modules

  10 * provided in the `context` parameter when creating the `Config` node. 

  11 * ```

  12 * {

  13 *  root: { Layout: {

  14 *      rows:  ["30px", "fill", "10px"],

  15 *      css: '.my-example',

  16 *      content: ['Header', 'Body', 'Footer]

  17 *  }},

  18 *  Header: { Layout:{

  19 *      columns: ["200px", "fill"],

  20 *      content: ['Left Header', 'Main Header']

  21 *  }},

  22 *  Body: { Layout:{

  23 *      columns: ["200px", "fill"], 

  24 *      content: ['LeftNav', 'MainNav']

  25 *  }},

  26 *  Footer: { Layout: {

  27 *      css: '.hs-site-footer',

  28 *      content: ['(c) Helpful ; Scripts']

  29 *  }},

  30 *  LeftNav:  { LeftNav:  { lib:"route.lib", field:"route.field"}},

  31 *  MainNav:  { MainNav:  { lib:"route.lib", field:"route.field"}}

  32 * }

  33 * ```

  34 * 

  35 * ### Calling pattern

  36 * Instantiate the component via

  37 * ```

  38 *  m(Config, { 

  39 *      source: src, // a file name, or obect literal

  40 *      context:[]   // a list of es6 modules against which to resolve components at runtime 

  41 *  });

  42 * ```

  43 * Additional parameters can be passed to the root node. The example below establishes routing and passes

  44 * `field` and `lib` to the root node:

  45 * ```

  46 *  class Router {

  47 *      view(node:Vnode) { 

  48 *          return m(Config, Object.assign({source: src, context: [hslayout, header, left, main]}, node.attrs));

  49 *      }

  50 *  }

  51 *  m.route(document.body, '/api', { 

  52 *      '/api':             Router,

  53 *      '/api/:lib':        Router,

  54 *      '/api/:lib/:field': Router

  55 *  });

  56 * ```

  57 */

  58

  59 /** */

  60import { m, Vnode } from './mithril'; 

  61

  62/**

  63 * `Component` class that creates a tree of mithril components out of a configuration obect or file.

  64 */

  65export class Config {

  66    oninit(node:Vnode) {

  67        const context:any[] = node.attrs.context;

  68        if (typeof node.attrs.source === 'string') { 

  69            if (!node.state.cfg) { m.request({ method: "GET", url: node.attrs.source }).then((s:any)=> {

  70                node.state.cfg = translate(s, s.root, context);

  71            }); }      

  72        } else {

  73            node.state.cfg = translate(node.attrs.source, node.attrs.source.root, context);

  74        }

  75    }

  76    view(node:Vnode) { 

  77        const cfg = node.state.cfg;

  78        return (cfg && cfg.compClass)? m(cfg.compClass, Object.assign(cfg.attrs, node.attrs)) : m('div', 'waiting');

  79    }

  80}

  82/**

  83 * recursively translates a configuration, trying to fetch the class definition for each element (key) in `config`. 

  84 * If successful, it creates an object literal containing the component class and its attributes.

  85 * If unsuccessful, the element's value is returned unaltered so that it can be consumed 

  86 * by an instance further up in the recursion tree.

  87 * @param config an object literal containing the entire configuration tree

  88 * @param subcfg an object literal containing a configuration subtree

  89 * @param context an array of objects against which to instantiate elements of `config`.

  90 * @return an object literal representing the configuration, with Class names resolved 

  91 * against the provided `context`.

  92 */

  93function translate(config:any, subcfg:any, context:any[]) {

  94    // resolve if synonym

  95    if (isSynonym(config, subcfg)) { subcfg = config[subcfg]; }

  96    // return if primitive

  97    if (['string', 'number', 'boolean', 'function'].indexOf(typeof subcfg)>=0) { return subcfg; }

  98    let result = subcfg.length? [] : {};

  99    // step through options 

 100    const options = Object.keys(subcfg);

 101    options.map((opt:string):Vnode => {

 102        const cl:any = resolve(opt, context);

 103        const content = translate(config, subcfg[opt], context); 

 104        if (cl) { 

 105            const r = {

 106                compClass:cl,   // Component class

 107                attrs:content   // attributes passed to the Component class

 108            };

 109            (!Array.isArray(subcfg) && options.length === 1)? 

 110                result = r : 

 111                result[opt] = r;   

 112        }

 113        else { result[opt] = content; }

 114    }); 

 115    return result; 

 116}

 118/**

 119 * resolves the symbol `sym` against the provided `context`.

 120 * If successful, returns the class definition for `sym`. 

 121 * @param sym the symbol to resolve

 122 * @param context the context to resolve against; `mithril` and `hsLayout` 

 123 * are implicitely part of the context and need not be specified.

 124 * @return the resolved Class, or `undefined`.

 125 */

 126function resolve(sym:string, context:any[]) {

 127    let cl:any;

 128    context.some((c:any) =>  cl = c[sym]);

 129    return cl;

 130}

 131

 132function isSynonym(config:any, key:any) { return typeof key === 'string' && config[key]; }
