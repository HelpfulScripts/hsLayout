   1/**

   2 * # Layout

   3 * A `mithril` component class that layouts available space in the window.

   4 * 

   5 * ### Invocation

   6 * invoked as `m(Layout, {:[], content:Array})`. See example below.

   7 * 

   8 * ### Attributes (node.attrs):

   9 * - ``: [...]: a required keyword indicating the {@link view.Layouter Layouter} to use, followed by a configuration array. 

  10 *    E.g. `rows: ["30px", "fill"]`.

  11 * - `content`: Array, required. The Vnode children to lay out. 

  12 *     - `string` literal, e.g. "the content"

  13 *     - `[ ]` array of elements:

  14 *         - `string` literal, e.g. "the content"

  15 *         - `{ }` literal

  16 *            - `compClass`: a {@link view.Layout.Component Component} class definition, e.g. `Layout`

  17 * - `css`: optional String. The css specifier to use for this `Layout` component.

  18 * - `route`: optional object literal holding parameters passed from `m.route`

  19 * - `href`: optional String. If present, makes the component clickable

  20 * - `target`: optional string, determines where `href` is opened. Defaults to '_blank' to open `href` in a new tab.

  21 * - `onclick`:(), optional. The function to call when clicked

  22 * 

  23 * ### Example

  24 * 

  25 * 'script.js'>

  26 * m.mount(root, {view: () => m(hsLayout.Layout, {

  27 *     css:     'myColumn',

  28 *     rows:    ["50px", "fill"], 

  29 *     content: ['Top row: 50px', 'Bottom row: remainder']

  30 *     })

  31 * });

  32 * 

  33 * 

  34

  35 */

  42/**

  43 * Component interface. 

  44 * Formalizes the `mithril` requirement for a `view` method.

  45 */

  46export interface Component {

  47    view(node:Vnode): Vnode;

  48}

  50/**

  51Base class for applying layouts. Subclasses should implement a {@link view.Layout.Layout.getComponents `getComponents`} method that returns

  52the components to render. The default implementation returns the conponents passed in `node.attrs.content`.

  53Optionally, the subclass can also implement {@link view.Layout.Layout.getCSS `getCSS`} to provide the CSS class to 

  54assign to the component, and override the default implementation, which returns `node.attrs.css`. 

  55### Example:

  56

  57import { Layout, px, FILL }  from 'hslayout';

  58const TitleHeight   = px(30); 

  59const FooterHeight  = px(10); 

  60class MyLayout extends Layout {

  61    getComponents(node:Vnode):Vnode {

  62        return this.layout('.my-layout', { rows:[TitleHeight, FILL, FooterHeight] }, [

  63            m(), 

  64            m(),

  65            m()

  66        ]);

  67    }

  68    getCSS(node:Vnode):string {

  69    }

  70} 

  71

  72 */

  73export class Layout implements Component {

  74    /**

  75     * holds structural elements in style form: left, right, top, bottom, width, height

  76     */

  77    public style:string;

  78

  79//    oninit(node:Vnode)   { this.report('Layout:init', node); }

  80//    oncreate(node:Vnode) { this.report('Layout:create', node); } 

  81//    onupdate(node:Vnode) { this.report('Layout:update', node); }

  82

  83    /**

  84     * Called during the lifecycle `view` call to retrieve the subcomponents to render in this container.

  85     * The default implementation returns components stored in `node.attrs.content`. This allows for 

  86     * creating containers directly via mithril: `m(Layout, {content:[...]})`. 

  87     * In case `node.attrs.content` is an array of literals with a `compClass` field describing a Component class, 

  88     * the method will create a Mithril node on that class and pass the `node.attrs.route` argument down to it.

  89     * 

  90     * Override this method to create containers that return more sophisticated content.

  91     * @return a String, a Vnode, or an array of Strings or Vnodes

  92     */

  93    protected getComponents(node:Vnode):Vnode {

  94        return !Array.isArray(node.attrs.content)? node.attrs.content :

  95            node.attrs.content.map((c:any) => {

  96                if (c.compClass) { 

  97                    c.attrs.route = node.attrs.route;

  98                    return m(c.compClass, c.attrs);

  99                } else {

 100                    return c;

 101                }

 102            });

 103    }

 104

 105    /**

 106     * Called during the lifecycle `view` call to retrieve the css style class to apply to this container.

 107     * The default implementation returns components stored in `node.attrs.css`. This allows for 

 108     * creating containers directly via mithril: `m(Layout, {content:[...], css:'.my-class'})`.

 109     * Override this method to create containers that return more sophisticated content.

 110     */

 111    protected getCSS(node:Vnode):string {

 112        return node.attrs.css || '';

 113    }

 114

 115    /**

 116     * ensures that all `components` are `mithril` nodes and returns them in an array.

 117     * @param components either of:

 118     * - a `string` literal --> interprets the string as trusted HTML and returns a leaf node.

 119     * - an array of elements, each either a `Layout`, or `string` literal which will be converted to a `Layout`.

 120     */

 121    private normalizeContent(components:Array|string): Vnode {

 122        if (typeof components === 'string') { 

 123            return [m('.hs-leaf', m.trust(components))]; 

 124        }

 125        if (components.length>0) { // an array: ensure elements are Layout components

 126            return components.map((comp:string|typeof Layout):Vnode => 

 127                (comp instanceof Layout)? comp : m(Layout, {content:comp})

 128            );

 129        }

 130        // else: assume components is a mithril node: return node as an array

 131        return [components];

 132    }

 133

 134    /**

 135    lays out the component in `components` according to the configuration in `attrs`.

 136    The method returns a vnode container that has an associated `cssClass` style.

 137    `layout` is called during the `render` phase of the `mithril` lifecycle, 

 138    which ensures an outside-in calling sequence on containers; 

 139    i.e. the outermost containers are called first, and `node` will already have the 

 140    `style` field set with required style attributes. 

 141    These are added to any `attrs` parameter provided.

 142

 143    The format for the layout configuration in `attrs` is 

 144    {: }

 145    

 146     where `keyword` is the keyword with which the `Layouter` was registered.

 147     * @param cssClass a css style designator; same as used in m(cssClass, ...) 

 148     * @param layout the attribute object literal that configures the layout

 149     * @param components the components to layout within the container. 

 150     * This is either a primitive `string`, or an array of `Layout`s or `string`s.

 151     * @return a vnode that has an associated `cssClass` style.

 152     */

 153    view(node:Vnode): Vnode {

 154        const content = this.normalizeContent(this.getComponents(node)); // --> Vnode[]

 155        let css = Layouter.createLayout(node.attrs, content);

 156        const attrs:any = {

 157            style: node.style,

 158            route: node.attrs.route,     

 159            onclick: node.attrs.onclick

 160        };

 161        node.attrs.route = undefined;

 162        if (node.attrs.href) { 

 163            log.debug(`href ${node.attrs.href}`);

 164            attrs.href = node.attrs.href;

 165            attrs.target = attrs.target || '_blank';

 166            attrs.oncreate = m.route.link;

 167            attrs.onupdate = m.route.link;

 168            // attrs.onclick = () => window.open(node.attrs.href, '_blank');

 169            return m(`a.hs-layout ${css} ${this.getCSS(node)}`, attrs, content.map((c:any) => c));

 170        } else {

 171            return m(`.hs-layout ${css} ${this.getCSS(node)}`, attrs, content.map((c:any) => c));

 172        }

 173    }
