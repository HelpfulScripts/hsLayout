  42## Attributes
  43The following options are supported for the Attributes array:
  44- **[ ]**: An empty array; The available tiles will cover the entire width and height. 
  45    Their size will adapt as tiles are added.
  46- **[FILL]**: like [ ], except the last tile fills the available space. 
  47- **[v]**: All components have the specified width and height (in px or %) and will fill the available space from the left,
  48    leaving any remaining unused space on the right. 
  49- **[w, h]**: All components have the specified width w and height h (in px or %) and will fill the available space from the left,
  50    leaving any remaining unused space on the right. 
  51- **[v, FILL]**: like [v], except that the last tile in each row will fill the remaining available width 
  52    and the tiles in the bottom row will fill the remaining height
  53- **[w, h, FILL]**: like [w, h], except that the last tile will fill the remaining available width. 
  54
  55v, w, and h can take on the following vakues:
  56- **px(n)** or **"_n_ px"** -- a fixed number of pixels

  58- **FILL** or **"fill"**   -- fills the remaining space - may appear only once per array.

  78    /**
  79     * Constructs a Tileds layout
  80     * @param areaDesc Description of the requested layout 
  81     */
  82    constructor(public areaDesc:LayoutToken[]) { 
  83        super(areaDesc); 
  84        // if any of the dimensions are in px, use the pixel method; else use the percent method
  85        // get unitPixel if any area is PixelToken           
  86        this.unit = areaDesc.some((area:LayoutToken) => (area instanceof PixelToken))? 
  87            this.unitPixel : this.unitPercent;           
  88    };
  89
  90    private unitPercent(num:number) {
  91        const desc = this.areaDesc;
  92        const fill = this.areaDesc.some(a => (a instanceof FillToken));
  93        const root = Math.sqrt(num);
  94        const rows = Math.round(root);
  95        let   cols = Math.floor(root);
  96        if (root > cols) { cols++; }
  97        let width  = (desc[0] instanceof DefinedToken)? desc[0].getSize() : undefined;
  98        let height = (desc[1] instanceof DefinedToken)? desc[1].getSize() : width;
  99
 100        width  = width  || 100/cols;
 101        height = height || 100/rows;
 102        let left = 0;
 103        let top  = 0;
 104
 105        let styles = [...Array(num).keys()].map(i => { 
 106            let r = 'auto';    let w = width+'%'; 
 107            let b = 'auto';    let h = height+'%';
 108            if ((left + 2*width) > 100 && fill) { r = '0%'; w = 'auto'; }
 109            if ((top + 2*height) > 100 && fill) { b = '0%'; h = 'auto'; }
 110            const style = `
 111                top: ${Math.floor(top)}%; bottom:${b};
 112                left: ${left}%;           right:${r};
 113                width: ${w};              height: ${h};
 114            `;
 115            if (Math.round(left += width) > 100-Math.floor(width)) { left = 0; top += height; }
 116            return style;
 117         });
 118        return styles;    
 119    };
 120
 121    private unitPixel(num:number) { // pattern: [px, px, FILL]
 122        const desc = this.areaDesc;
 123//        const fill = this.areaDesc.some(a => (a instanceof FillToken));
 124        const root = Math.sqrt(num);
 125        const rows = Math.round(root);
 126        let   cols = Math.floor(root);
 127        if (root > cols) { cols++; }
 128        let width  = (desc[0] instanceof DefinedToken)? desc[0].getSize() : undefined;
 129        let height = (desc[1] instanceof DefinedToken)? desc[1].getSize() : width;
 130
 131        width  = width  || 100/cols;
 132        height = height || 100/rows;
 133        let left = 0;
 134        let top  = 0;
 135
 136        let styles = [...Array(num).keys()].map(i => { 
 137            let r = 'auto';    let w = width+'px'; 
 138            let b = 'auto';    let h = height+'px';
 139            const style = `
 140                top: ${Math.floor(top)}%; bottom:${b};
 141                left: ${left}%;           right:${r};
 142                width: ${w};              height: ${h};
 143            `;
 144            if (Math.round(left += width) > 100-Math.floor(width)) { left = 0; top += height; }
 145            return style;
 146         });
 147        return styles;    
 148    };

 150    /**
 151     * Calculates the style attributes required for each component in `Components`.
 152     * These attributes are saved in a `styles` field on the component itself.

 160        });
