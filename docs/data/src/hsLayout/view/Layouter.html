  20

  21class MyLayouter extends Layouter {

  22    cssClass = '.my-css-class';

  23    constructor(public areaDesc:LayoutToken[]) { 

  24        super(areaDesc); 

  25    }

  26    

  27    protected getStyles(components:Array):string {

  28        components.map((c:Layout|Vnode, i:number) => {

  29            c.style = `width:auto; height:auto;

  30        });   

  31        return this.cssClass;

  32   }

  33}

  38export abstract class Layouter {

  39    /**

  40     * statis list of available styles. The key for each entry is the keyword that triggers the style,

  41     * and the value is a constructor for that style

  42     */

  43    static layoutStyles:{string?: Layouter} = {};

  45    /**

  46     * Translates `string` params to {@link hsLayout:view.Tokens.LayoutToken LayoutTokens}.

  47     * The `params` are expected to either

  48     * - end in 'px'

  49     * - end in '%'

  50     * - be equal to 'fill' (case insensitive)

  51     * @param params an Array of strings that will be converted to an array of LayourTokens.

  52     * 

  53     */

  54    private static translate(params:Array):Array {

  55        if (params.length === 0) { params.push(''); }   //  [] --> ['']

  56        return params.map((param:string|any) => {

  57            if (typeof param === 'string') {

  58                if (param.endsWith('px')) { return px(parseInt(param)); }

  59                if (param.endsWith('%')) { return pc(parseInt(param)); }

  60                if (param.toLowerCase()==='fill') { return FILL;}

  61            } else {

  62                return param;

  63            }

  64        });

  65    }

  67    /**

  68     * Register a new Layouter style with corresponding configuration keyword.

  69     * Example:```

  70     * class ColumnsLayout extends Layouter {...}

  71     * Layouter.register('Column', Columns);

  72     * ```

  73     * @param keyword the keyword used in the attributes to `this.layout`

  74     * @param style the `Layouter` implementation to instantiate when encountering `keyword` 

  75     */

  76    public static register(keyword:string, style:typeof Layouter) {

  77//        console.log(`registering ${keyword} layout`);

  78        Layouter.layoutStyles[keyword] = style;

  79    }

  80

  81    /**

  82     * Lays out the `components` according to the configuration in `attrs`.

  83     * The method will search for a registered layout key in `attrs`, then construct the `Layouter` associated with the key

  84     * with the parameters for the key, and call the `getStyles` method on the style with the provided `components`.

  85     * @param attrs an object literal, typically provided as middle attributes object in the m(css, {}, '') call.

  86     * @param components 

  87     * @return returns the css class that the `getStyles` function returns.

  88     */

  89    public static createLayout(attrs:any, components:Array):string {

  90        let css = '';

  91        Object.keys(Layouter.layoutStyles).some(key => { // executes the first match key in attrs.

  92            if (attrs[key]) { 

  93                css = new Layouter.layoutStyles[key](Layouter.translate(attrs[key])).getStyles(components); 

  94                attrs[key] = undefined;

  95                return true;

  96            }

  97            return false;

  98        });

  99        return css;

 100    }

 103    spacing = 0;

 107     * The `areaDesc` parameter is expected to be of the form {: {@link hsLayout:view.Tokens.LayoutToken `LayoutToken`}[]}} 

 108     * and will be passed through form the `Layout` requesting the layout.

 121        });

 127    protected abstract getStyles(components:Array):string;
