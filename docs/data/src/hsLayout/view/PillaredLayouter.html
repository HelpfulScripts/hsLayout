   1/**
   2Lays out components in pillars, i.e. either {@link hsLayout:PillaredLayout.Columns columns}
   3or {@link hsLayout:PillaredLayout.Rows rows}
   4Use either of
   5- `{rows: [attributes]}`
   6- `{columns: [attributes]}` 
   7
   8to invoke this layout.
   9
  10### Example
  11
  12    {rows: [px(200), FILL]}   // --> top row has height 200px, all other rows evenly share remaining space 
  13    {rows: ["200px", "fill"]} // --> equivalent description
  14
  15
  16## Attributes
  17The following values **v** are valid entries in The Attributes array:
  18- **px(n)** or **"px"** -- a fixed number of pixels 
  19- **pc(n)** or **"%"**  -- a fixed percentage of available space
  20- **FILL** or **"fill"**   -- a special constant to indicate - may appear only once per array.
  21
  22The following options are supported for the Attributes array:
  23- **[ ]**: An empty array; all components will be evenly spaced across the available width. 
  24- **[v]**: All components have the specified width (in px or %) and will fill the available space from the left,
  25    leaving any remaining unused space on the right. 
  26- **[v1, v2]**: All components have the specified widths (in px or %) and will fill the available space from the left,
  27    leaving any remaining unused space on the right. If there are more components than widths, the right-most width
  28    will be used for the reminaing widgets.
  29- **[v, FILL]**: Sets the first (left) widget to a width of `v`.

  30    if `v` is specified in %, the remaining n-1 components will have equal relative widths of `(100-v)/(n-1)%`

  31    if `v` is specified in px, the remaining n-1 components will have their right borders at location `i*100/n%`, with i=1...n.
  32- **[FILL, v]**: Sets the last (right) widget to a width of `v`.

  33    if `v` is specified in %, the remaining n-1 components will have equal relative widths of `(100-v)/(n-1)%`

  34    if `v` is specified in px, the remaining n-1 components will have their left borders at location `i*100/n%`, with i=0...n-1.
  35- **[va, FILL, vb]**: Sets the first and last widget to a width of `va`/`vb`.

  36    Both have to be specified either in px or in %.

  37    if the unit is %, the remaining n-2 components will have equal relative widths of `(100-vb-va)/(n-2)%`

  38    if the unit is px, the remaining n-2 components will have their left/right borders at location `i*100/n%`.
  39- **[v1, v2, FILL, v3, v4]**: multiple widths can be specified in uninterrupted sequence both from the left and the right. 
  40 */

  83/**
  84 * Abstract base Layouter for creating PillarLayouter (rows, colums)
  85 */
  86abstract class PillarLayouter extends Layouter{
  87    firstFixed: number; // number of DefinedToken entries at the beginning
  88    lastFixed:  number; // number of DefinedToken entries at the end
  89    unit:(num:number)=>descriptor[];
  90    fields: string[];
  91    cssClass:string;
  92
  93    /**
  94     * Constructs a Pillared layout (rows or columns).
  95     * Determines the `unit` (% or px) from the passed area descriptors
  96     * @param params Style params for either rows or columns layout
  97     * @param areaDesc Description of the requested layout 
  98     */
  99    constructor(params:PillarParams, public areaDesc:LayoutToken[]) { 
 100        super(areaDesc); 
 101        this.fields = params.fields;
 102        this.cssClass = params.cssClass;
 103
 104        let n = areaDesc.length-1;
 105        let first = 0;
 106        let last  = 0;        
 107        // if any of the dimensions are in px, use the pixel method; else use the percent method
 108        this.unit = areaDesc.some((area:LayoutToken) => (area instanceof PixelToken))? 
 109            this.unitPixel : this.unitPercent;   
 110        
 111        // determine first = number of consecutive fixed fields at start
 112        areaDesc.some((area:LayoutToken, i:number) => 
 113            ((areaDesc[i]   instanceof DefinedToken)? ++first<0 : true));         
 114
 115        // determine last  = number of consecutive fixed fields at end
 116        areaDesc.some((area:LayoutToken, i:number) => 
 117            ((areaDesc[n-i] instanceof DefinedToken)? ++last<0  : true));         
 118
 119        this.firstFixed = first;
 120        this.lastFixed  = Math.min(last, areaDesc.length-first);
 121    };
 122
 123    // num: number of areas to layout
 124    /**
 125     * Creates an iterable list of size descriptors, one for each area to be layed out.
 126     * Each descriptor 
 127     * @param num the number of areas to be layed out
 128     * @return Iterable list of `num` size descriptors, one for each area to be layed out
 129     */
 130    private getSizes(num:number):descriptor[] {
 131        const first = this.firstFixed;
 132        const last  = this.lastFixed;
 133        const desc  = this.areaDesc;
 134        const len = desc.length;
 135
 136        return [...Array(num).keys()].map((i:number) => {
 137            let size:number = null;
 138            let t = null;
 139            if (i > num-1-last)  { size = desc[len - (num-i)].getSize(); t = 'end'; }       // end sequence
 140            else if (i < first)  { size = desc[i].getSize(); t = 'start'; }                 // start sequence
 141            else if (len>0 && len===first){ size = desc[len-1].getSize(); t = 'start'; }    // all items 
 142            return {size:size, code:t, fields:{}};
 143        });
 144    }
 145
 146    private unitPercent(num:number):descriptor[] {
 147        let f = this.fields;
 148        let max = 100.0;
 149        let styles:descriptor[] = this.getSizes(num);
 150
 151        styles.forEach(style => { if (style.size) { max = max - style.size; num--; } });
 152        let defDim = max / num;      // divvy up remaining space
 153
 154        function pass(styles:descriptor[], ix0:string, ix1:string, breakCond:(cond:string)=>boolean) {
 155            let sumDim = 0;
 156            styles.some(style => { // stop when breakCond is met
 157                let size = style.size || defDim;
 158                if (breakCond(style.code)) { return true; }
 159                style.fields[ix0] = sumDim+'%';
 160                sumDim += size;
 161                style.fields[ix1] = (100-sumDim)+'%'; 
 162                style.fields[f[5]] = 'auto';
 163                return false;
 164            });
 165        }
 166
 167        pass(styles, f[2], f[3], (e:string) => e==='end');              // forward pass
 168        pass(styles.reverse(), f[3], f[2], (e:string) => e!=='end');    // backward pass
 169        return styles.reverse();    // reverse a second time for original sequence.
 170    };
 171
 172    private unitPixel(num:number):descriptor[] { // pattern: [px, px, FILL , px, px]
 173        let styles:descriptor[] = this.getSizes(num);
 174        let f = this.fields;
 175       
 176        let defDim = 100.0/num;          // used for unspecified widths
 177
 178        // work forwards through the heights
 179        let sumDim = 0;
 180        styles.some((style, i) => {
 181            if (style.code==='start') {   // so far, all heights explicitly set as px
 182                style.fields[f[2]] = sumDim +'px';
 183                sumDim += style.size + (this.spacing || 0) + (this.spacing || 0);
 184                style.fields[f[3]] = 'auto';
 185                style.fields[f[5]] = style.size +'px';
 186            } else if (style.code === null) {
 187                style.fields[f[2]] = (sumDim>0)? (sumDim +'px') : (i*defDim + '%');
 188                sumDim = -1;
 189                style.fields[f[3]] = (100-(i+1)*defDim) + '%';
 190                style.fields[f[5]] = 'auto';
 191            } else if (style.code==='end') { return true; }
 192            return false;
 193        });
 194        
 195        // work backwards through the heights
 196        sumDim = 0;
 197        styles.slice().reverse().some((style, i) => {
 198            style.fields[f[3]] = sumDim + 'px';
 199            if (style.code === 'end') { 
 200                sumDim += style.size + (this.spacing || 0) + (this.spacing || 0);
 201                style.fields[f[2]] = 'auto';
 202                style.fields[f[5]] = style.size+'px';
 203            } else {
 204                if (sumDim>0 && style.code !== 'start') {
 205                    style.fields[f[5]] = 'auto';
 206                }
 207                return true; 
 208            } 
 209            return false;
 210        });  
 211        return styles;
 212    };

 214    /**
 215     * Calculates the style attributes required for each component in `Components`.
 216     * These attributes are saved in a `styles` field on the component itself. 
 217     * During rendering these `styles` attributes are copied to the `node.attrs.styles` field.
 218     * @param components 
 219     */
 220    protected getStyles(components:Array):string  { 
 221        let f = this.fields;
 222        let styles:descriptor[] = this.unit(components.length);
 223        components.map((c:Layout|Vnode, i:number) => {
 224            c.style = `${f[0]}:0%; ${f[1]}:0%; `;
 225            Object.keys(styles[i].fields).forEach((st:string) => { c.style += `${st}: ${styles[i].fields[st]};`; });
 226        });   
 227        return this.cssClass;
 228    };
