Class HsConfig

Interprets a configuration and either mounts it or routs it in mithril. Example: import * as mylib from './mylib';

const myConfig = { Layout: { rows: ['50px', 'fill'], css: '.hs-site', content: [{MyClass: {}}, 'bottom row'] }, route: { default: '/api', paths: [ '/api', // defines http://localhost/#!/api/ '/api/:lib', // defineshttp://localhost/#!/api/foo '/api/:lib/:field' // defines `http://localhost/#!/api/foo/bar ] } }

const myExample = { MyClass: class extends Layout { view(node:Vnode) { return m('', 'myExample'); } } }

new HsConfig([mylib, myExample]).attachNodeTree(myConfig, document.body)
